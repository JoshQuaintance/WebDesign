Normal JavaScript DOM
//Without jQuery
var Playlist = document.getElementById("spotify");

function spotifyPlaylist() {
	if (Playlist.style.display === "none") {
	Playlist.style.display = "inline";
	} else {
	Playlist.style.display = "none";
	}
}
	
Javascript With jQuery
//Normal Javascript 
function spotifyPlaylist() {
	//if the playlist is displayed, then run this
	if ($("#spotify").css("display") === "none") {
		$("#spotify").css("display", "inline")
	} //otherwise run this one instead 
		else {
		$("#spotify").css("display", "none")
	}
}
